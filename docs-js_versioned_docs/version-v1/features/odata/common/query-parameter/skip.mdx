import Tabs from '@theme/Tabs';
import TabItem from '@theme/TabItem';

`skip` allows you to skip a number of results in the requested set.
It can be useful for paging:

<Tabs groupId="version" defaultValue="major" values={[ { label: 'SDK 2.x', value: 'major' }, { label: 'SDK 1.x', value: 'deprecated' } ]}>
<TabItem value="major">

```ts
const { businessPartnerApi } = businessPartnerService();
businessPartnerApi.requestBuilder().getAll().skip(10);
```

</TabItem>

<TabItem value="deprecated">

```ts
BusinessPartner.requestBuilder().
  .getAll()
  .skip(10)
```

</TabItem>
</Tabs>
